n 个整数的无序数组，找到每个元素后面比它大的第一个数，要求时间复杂度为 O(N)

```cpp
vector<int> findmax(vector<int> num)
{
    if (num.size() == 0)
    {
        return num;
    }
    vector<int> result(num.size());
    int i = 0;
    stack<int> s;
    while (i < num.size())
    {
        if (s.empty() || num[s.top()] >= num[i])
        {
            s.push(i++);
        }
        else
        {
            result[s.top()] = num[i];
            s.pop();
        }
    }
    while (!s.empty())
    {
        result[s.top()] = INT_MAX;
        s.pop();
    }
    for (int i = 0; i < result.size(); ++i)
    {
        std::cout << result[i] << std::endl;
    }
    return result;
}
```
